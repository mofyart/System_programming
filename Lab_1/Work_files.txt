# Создаем файл Work_files и записываем все команды
{
echo "=== КОМАНДЫ ДЛЯ РАБОТЫ С ФАЙЛАМИ ==="

# 1. Создаем структуру папок в файле Work
mkdir Лабораторная работа №1!
mkdir Бусыгин Артём Дмитриевич

# 2. Устанавливаем права на папку Лабораторная работа №1!
chmod 774 Work/Лабораторная\ работа\ №1\!

# 3. Устанавливаем права на папку 'ФИО'"\
chmod 200 Work/Бусыгин\ Артём\ Дмитриевич

# 4. Создаем три файла в папке Work
touch Text@1
touch Text#2
touch Text#3

# 5. Записываем текст в файл 'Text @1' без редактора
echo "Птица- говорун отличается умом и сообразительностью\!" > Text\@1.txt
Следующая строка
echo "Отличается умом, отличается сообразительностью..." >> Text\@1.txt

#6 Без использования текстового редактора скопируете содержимое файла «Text @1» в файл «Text $2».
cat Text\@1.txt > Text\$2.txt

#7 Переместите файл «Text $2» в папку «Лабораторная работа №1!» и скопировать файл «Text @1» в папку «ФИО».
Меняем права доступа для копирования: chmod 300 "Бусыгин Артём Дмитриевич"
Копируем cp "Text@1.txt" "Бусыгин Артём Дмитриевич/"

#8 Сожмите файлы «Text @1», «Text #3» с помощью утилиты gzip и добавьте их в архив «MyAchiv» с помощью утилиты tar
gzip "Text@1.txt"
gzip "Text#3.txt"
tar -cvf "MyArchive.tar" "Text@1.gz" "Text#3.gz"

#9 Добавьте в файл «Text #3» первую строку из файла «Text @1» без использования текстового редактора,
а далее в конец «Text #3» добавьте строку «Будь осторожен! Преступник вооружен!»
head -n 1 Text\@1.txt > Text\#3.txt
echo "Будь осторожен\! Преступник вооружен\!" >> Text\#3.txt

#10 В папке «ФИО» создайте файл «Result_one», в который запишите в обратном порядке содержимое файла «Text @1»
tac Text\@1.txt > "Бусыгин Артём Дмитриевич/Result_one.txt"

#11 Переименуйте файл «Result_one» в «Result_two»
mv "Бусыгин Артём Дмитриевич/Result_one.txt" "Бусыгин Артём Дмитриевич/Result_two.txt"

#12 В папке «Лабораторная работа №1!» создайте файл «Result_3», в который запишите содержимое файлов и папок папки «ФИО»,
начинающихся в своем названии с буквы «T».
cat Бусыгин\ Артём\ Дмитриевич/T* > Лабораторная\ работа\ №1\!/Result_3.txt

#13 Добавьте, без использования текстового редактора, в конец файла «Result_3» полную информацию о ядре системы и текущую дату.
uname -a >> Лабораторная\ работа\ №1\!/Result_3.txt - полные данные о ядре
date >> Лабораторная\ работа\ №1\!/Result_3.txt - дата
}
